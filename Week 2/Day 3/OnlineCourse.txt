CLASS Course
    PRIVATE title, duration, price
    CONSTRUCTOR(title, duration, price)
    GETTERS for title, duration, price

CLASS Student
    PRIVATE name, email, enrolledCourses (List<Course>)
    CONSTRUCTOR(name, email)
    
    METHOD enrollCourse(course)
        IF course NOT IN enrolledCourses
            ADD course TO enrolledCourses
            PRINT name + " enrolled in " + course.title
        ELSE
            PRINT "Already enrolled in " + course.title

    METHOD viewEnrolledCourses()
        PRINT name + "'s Enrolled Courses:"
        FOR EACH course IN enrolledCourses
            PRINT course.title + " - " + course.duration + " - â‚¹" + course.price
