FUNCTION merge(left, right)
    result = []
    WHILE left NOT EMPTY AND right NOT EMPTY
        IF left[0].marks > right[0].marks
            ADD left[0] TO result
            REMOVE left[0]
        ELSE IF left[0].marks == right[0].marks
            IF left[0].name < right[0].name
                ADD left[0]
                REMOVE left[0]
            ELSE
                ADD right[0]
                REMOVE right[0]
        ELSE
            ADD right[0]
            REMOVE right[0]

    ADD remaining left, right TO result
    RETURN result
