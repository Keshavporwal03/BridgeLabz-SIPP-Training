CLASS Transaction<T>
    PRIVATE paymentDetails
    CONSTRUCTOR(paymentDetails)
        this.paymentDetails = paymentDetails
    
    METHOD processPayment()
        IF paymentDetails IS UPI
            PRINT "Payment of ₹" + paymentDetails.amount + " via UPI successful for " + paymentDetails.merchantName
        ELSE IF paymentDetails IS Card
            PRINT "Payment of ₹" + paymentDetails.amount + " via Card ending " + last4Digits(paymentDetails.cardNumber) + " successful for " + paymentDetails.merchantName
        ELSE IF paymentDetails IS NetBanking
            PRINT "Payment of ₹" + paymentDetails.amount + " via NetBanking (" + paymentDetails.bankName + ") successful."

CLASS UPI
    upiId, amount, merchantName

CLASS Card
    cardNumber, cvv, expiryDate, amount, merchantName

CLASS NetBanking
    bankName, accountNumber, amount
